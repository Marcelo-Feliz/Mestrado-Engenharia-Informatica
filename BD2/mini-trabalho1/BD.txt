Create table avaliacao(

NumS char(3),

ISBN char(9),

Nestrelas int,

Comentario varchar(50),


primary key (NumS,ISBN),

foreign key (ISBN) references livro(ISBN) on  delete restrict,

foreign key (NumS) references socio on  delete restrict,

CHECK ( Nestrelas < 6),

CHECK ( Nestrelas > 0)

);
